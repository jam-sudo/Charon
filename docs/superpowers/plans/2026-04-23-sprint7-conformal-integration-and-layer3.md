Inside `class ConformalPredictor`, add classmethod:

```python
    @classmethod
    def load_default(
        cls,
        cache_path: Path | None = None,
        reference_csv: Path | None = None,
        clint_oof_path: Path | None = None,
        force_recalibrate: bool = False,
    ) -> "ConformalPredictor":
        """Return a calibrated predictor, using an on-disk cache when valid.

        The cache stores quantiles for ``fup`` (via ``calibrate`` over
        ``adme_reference.csv``) and ``clint_hepatocyte`` (via
        ``calibrate_from_oof`` on the persisted scaffold-CV residuals).
        Invalidated when any source file's SHA-256 changes.

        Args:
            cache_path: JSON cache location. Defaults to
                ``models/conformal_cache.json``.
            reference_csv: Calibration CSV for fup. Defaults to repo's
                ``data/validation/adme_reference.csv``.
            clint_oof_path: ``.npy`` file of |log10(pred/obs)| residuals
                produced by ``scripts/train_clint.py``.
            force_recalibrate: If True, ignore any existing cache.
        """
        cache_path = Path(cache_path) if cache_path else _DEFAULT_CACHE_PATH
        reference_csv = Path(reference_csv) if reference_csv else _DEFAULT_REFERENCE_CSV
        clint_oof_path = Path(clint_oof_path) if clint_oof_path else _DEFAULT_CLINT_OOF

        sources = {
            "adme_reference.csv": reference_csv,
            "xgboost_clint_oof_residuals.npy": clint_oof_path,
        }
        current_hashes = {
            name: _sha256(p) for name, p in sources.items() if p.exists()
        }

        if not force_recalibrate and cache_path.exists():
            try:
                cached = _json.loads(cache_path.read_text())
            except (ValueError, OSError) as exc:
                logger.warning(
                    "Corrupted conformal cache at %s (%s); falling back to calibration",
                    cache_path, exc,
                )
                cached = None
            if (
                cached is not None
                and cached.get("schema_version") == _CACHE_SCHEMA_VERSION
                and cached.get("source_hashes") == current_hashes
            ):
                cp = cls(calibration_data=reference_csv)
                for prop, rpt in cached["reports"].items():
                    cp._reports[prop] = CoverageReport(
                        property_name=prop,
                        n_samples=int(rpt["n_samples"]),
                        quantile_log10=float(rpt["quantile_log10"]),
                        factor=float(rpt["factor"]),
                        empirical_coverage=float(rpt["empirical_coverage"]),
                        median_fold_error=float(rpt["median_fold_error"]),
                        mean_fold_error=float(rpt["mean_fold_error"]),
                        warning=rpt.get("warning"),
                    )
                logger.info("Loaded conformal cache from %s", cache_path)
                return cp

        # Cache miss / hash mismatch / corrupted cache — recalibrate.
        cp = cls(calibration_data=reference_csv)
        predictor = ADMETPredictor()
        cp.calibrate(predictor)
        if clint_oof_path.exists():
            residuals = np.load(clint_oof_path)
            cp.calibrate_from_oof("clint_hepatocyte", residuals)
        else:
            logger.warning(
                "CLint OOF residuals missing at %s; clint_hepatocyte will not be calibrated. "
                "Run scripts/train_clint.py to generate them.",
                clint_oof_path,
            )
        cp._save_cache(cache_path, current_hashes)
        return cp

    def _save_cache(self, cache_path: Path, source_hashes: Mapping[str, str]) -> None:
        payload = {
            "schema_version": _CACHE_SCHEMA_VERSION,
            "generated_utc": datetime.now(timezone.utc).isoformat(),
            "source_hashes": dict(source_hashes),
            "coverage_target": self._coverage,
            "reports": {
                prop: {
                    "n_samples": rpt.n_samples,
                    "quantile_log10": rpt.quantile_log10,
                    "factor": rpt.factor,
                    "empirical_coverage": rpt.empirical_coverage,
                    "median_fold_error": rpt.median_fold_error,
                    "mean_fold_error": rpt.mean_fold_error,
                    "warning": rpt.warning,
                }
                for prop, rpt in self._reports.items()
            },
        }
        cache_path.parent.mkdir(parents=True, exist_ok=True)
        cache_path.write_text(_json.dumps(payload, indent=2))
        logger.info("Wrote conformal cache to %s", cache_path)
```

class _ConformalOff:

    global _default_conformal_instance

def render_adme_table(entries: list[dict]) -> str:
    header = "| property | value | 90% CI | unit | source |\n"

        )
    return header + "\n".join(rows) + "\n"

def _mc_sample(entry, n: int, seed: int = 42) -> dict[str, tuple[float, float, float]]:
    """Draw n log-space samples of fup and clint from their CIs, return p05/p50/p95 of CL/Vss."""
    import numpy as np

    rng = np.random.default_rng(seed)
    fup_prop = entry.compound.properties.binding.fu_p
    clint_prop = entry.compound.properties.metabolism.clint_uL_min_mg
    # Only sample if source == ml_ensemble AND CI present.
    samples_fup = _lognormal_samples(fup_prop, n, rng) if _is_ml(fup_prop) else None
    samples_clint = _lognormal_samples(clint_prop, n, rng) if _is_ml(clint_prop) else None
    if samples_fup is None and samples_clint is None:
        return {}

    cls, vsss = [], []
    base_props = entry.compound.properties
    for i in range(n):
        props_i = _perturb(base_props, fup=samples_fup[i] if samples_fup is not None else None,
                                        clint=samples_clint[i] if samples_clint is not None else None)
        pipe_i = Pipeline(
            entry.compound.model_copy(update={"properties": props_i}),
            route=entry.route, dose_mg=entry.dose_mg, duration_h=entry.duration_h,
        )
        res = pipe_i.run()
        cls.append(res.pk_parameters.cl_L_h)
        vsss.append(res.pk_parameters.vss_L)

    return {
        "cl_p05": float(np.percentile(cls, 5)),
        "cl_p50": float(np.percentile(cls, 50)),
        "cl_p95": float(np.percentile(cls, 95)),
        "vss_p05": float(np.percentile(vsss, 5)),
        "vss_p50": float(np.percentile(vsss, 50)),
        "vss_p95": float(np.percentile(vsss, 95)),
    }


def _is_ml(prop) -> bool:
    return prop is not None and getattr(prop, "source", None) == "ml_ensemble" \
           and prop.ci_90_lower is not None and prop.ci_90_upper is not None


def _lognormal_samples(prop, n: int, rng):
    import math
    import numpy as np
    mu = math.log10(prop.value)
    sigma_log = (math.log10(prop.ci_90_upper) - math.log10(prop.ci_90_lower)) / (2 * 1.645)
    return 10.0 ** rng.normal(loc=mu, scale=sigma_log, size=n)


def _perturb(props, *, fup=None, clint=None):
    """Return a copy of CompoundProperties with fup/clint perturbed."""
    binding = props.binding
    metabolism = props.metabolism
    updates = {}
    if fup is not None:
        updates["binding"] = binding.model_copy(update={
            "fu_p": binding.fu_p.model_copy(update={"value": float(fup)}),
        })
    if clint is not None:
        updates["metabolism"] = metabolism.model_copy(update={
            "clint_uL_min_mg": metabolism.clint_uL_min_mg.model_copy(update={"value": float(clint)}),
        })
    return props.model_copy(update=updates) if updates else props

```python
"""Layer 3 FIH dose benchmark — two-tier (gold + sanity-floor).

See docs/superpowers/specs/2026-04-23-sprint7-conformal-integration-and-layer3-design.md §6.
"""
from __future__ import annotations

import argparse
import math
import sys
from datetime import datetime, timezone
from pathlib import Path

import yaml

REPO_ROOT = Path(__file__).resolve().parents[2]
sys.path.insert(0, str(REPO_ROOT))

from charon import Pipeline  # noqa: E402
from charon.core.compound_config import load_compound_yaml  # noqa: E402
from charon.core.schema import DoseProjectionConfig  # noqa: E402
from validation.benchmarks.report_writer import emit_report  # noqa: E402

DEFAULT_PANEL = REPO_ROOT / "validation" / "data" / "fih_reference" / "panel.yaml"
COMPOUNDS_DIR = REPO_ROOT / "validation" / "data" / "tier1_obach" / "compounds"
DEFAULT_MD = REPO_ROOT / "validation" / "reports" / "layer3_fih_dose.md"
DEFAULT_JSON = REPO_ROOT / "validation" / "reports" / "layer3_fih_dose.json"


def fold_error(a: float, b: float) -> float:
    if a <= 0 or b <= 0:
        return math.inf
    return max(a / b, b / a)


def passes_floor(predicted_mg: float, approved_starting_mg: float) -> bool:
    return predicted_mg <= approved_starting_mg + 1e-9


def _compute_mrsd(entry: dict) -> tuple[float, str]:
    """Return (MRSD in mg, limiting_method) via the PAD path.

    Uses ``target_ceff_nM`` from the panel entry — PAD translates target
    steady-state exposure into a daily dose, then applies safety_factor=10.
    """
    compound = load_compound_yaml(COMPOUNDS_DIR / f"{entry['name']}.yaml")
    pipe = Pipeline(
        compound,
        route=entry["route"],
        dose_mg=1.0,  # seed dose for PK simulation; MRSD derives from PAD
        dose_projection=DoseProjectionConfig(
            target_ceff_nM=float(entry["target_ceff_nM"]),
            safety_factor=10.0,
            tau_h=24.0,
        ),
    )
    result = pipe.run()
    if result.dose_recommendation is None:
        raise RuntimeError(f"No dose recommendation for {entry['name']}")
    return float(result.dose_recommendation.mrsd_mg), result.dose_recommendation.limiting_method


def run_panel(panel_path: Path) -> dict:
    panel = yaml.safe_load(panel_path.read_text())["panel"]
    gold_rows: list[dict] = []
    floor_rows: list[dict] = []
    for entry in panel["compounds"]:
        mrsd_mg, limiting = _compute_mrsd(entry)
        if entry["tier"] == "gold":
            ref = entry["reference_fih_mg"]
            fold = fold_error(mrsd_mg, ref)
            gold_rows.append({
                "compound": entry["name"],
                "route": entry["route"],
                "mrsd_pred_mg": mrsd_mg,
                "limiting_method": limiting,
                "reference_fih_mg": ref,
                "source_type": entry.get("source_type", "unknown"),
                "fold_error": fold,
                "within_3x": fold <= 3.0,
                "within_10x": fold <= 10.0,
                "source": entry["source"],
            })
        else:
            start = entry["approved_starting_dose_mg"]
            floor_rows.append({
                "compound": entry["name"],
                "route": entry["route"],
                "mrsd_pred_mg": mrsd_mg,
                "limiting_method": limiting,
                "approved_starting_dose_mg": start,
                "pass_floor": passes_floor(mrsd_mg, start),
                "source": entry["source"],
            })
    n_gold = len(gold_rows)
    n_floor = len(floor_rows)
    summary = {
        "gold": {
            "n": n_gold,
            "within_3x": sum(r["within_3x"] for r in gold_rows),
            "within_3x_fraction": (
                sum(r["within_3x"] for r in gold_rows) / n_gold if n_gold else 0.0
            ),
            "within_10x": sum(r["within_10x"] for r in gold_rows),
        },
        "sanity_floor": {
            "n": n_floor,
            "pass_count": sum(r["pass_floor"] for r in floor_rows),
            "pass_fraction": (
                sum(r["pass_floor"] for r in floor_rows) / n_floor if n_floor else 0.0
            ),
            "failures": [r["compound"] for r in floor_rows if not r["pass_floor"]],
        },
    }
    return {
        "title": "Charon Layer 3 FIH Dose Benchmark",
        "panel": panel["name"],
        "date_utc": datetime.now(timezone.utc).isoformat(),
        "summary": {
            "gold_n": summary["gold"]["n"],
            "gold_within_3x": summary["gold"]["within_3x"],
            "gold_within_3x_fraction": summary["gold"]["within_3x_fraction"],
            "gold_within_10x": summary["gold"]["within_10x"],
            "sanity_n": summary["sanity_floor"]["n"],
            "sanity_pass_count": summary["sanity_floor"]["pass_count"],
            "sanity_pass_fraction": summary["sanity_floor"]["pass_fraction"],
            "sanity_failures": ",".join(summary["sanity_floor"]["failures"]) or "-",
        },
        "rows": [],  # benchmark splits into two tiers — see extra_sections
        "extra_sections": {
            "Gold (Tier A) — fold-error vs reference FIH": gold_rows,
            "Sanity floor (Tier B) — MRSD <= approved starting dose": floor_rows,
        },
        "notes": [
            "Tier A (gold): fold-error vs published FIH or approved-start dose. Report-only.",
            "Tier B (sanity_floor): MRSD_pred <= approved_starting_dose_mg. GATED — any failure blocks release.",
            "MRSD computed via PAD path (target_ceff_nM) with safety_factor=10.",
            "Sources: inline per compound in panel.yaml.",
        ],
    }


def main() -> int:
    ap = argparse.ArgumentParser()
    ap.add_argument("--panel", type=Path, default=DEFAULT_PANEL)
    ap.add_argument("--output-stem", type=Path,
                    default=DEFAULT_MD.with_suffix(""),
                    help="Output path stem; writes {stem}.md and {stem}.json")
    args = ap.parse_args()

    payload = run_panel(args.panel)
    emit_report(payload, stem=args.output_stem)

    # Gate on sanity floor.
    n_fail = len(payload["summary"]["sanity_floor"]["failures"])
    if n_fail:
        print(f"[FAIL] Sanity floor: {n_fail} compound(s) exceed approved start dose.",
              file=sys.stderr)
        return 2
    print(f"[OK] Sanity floor: {payload['summary']['sanity_floor']['pass_count']}/"
          f"{payload['summary']['sanity_floor']['n']} pass.")
    return 0


if __name__ == "__main__":
    sys.exit(main())
```
